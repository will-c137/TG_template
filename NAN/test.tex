\section{测试算法伪代码}
\begin{algorithm}
    \caption{求解$\int_{0}^{1}x^n \mathrm{e}^x \,  \mathrm{d}x$并计算相对误差}
    \label{alg:2}
    \begin{algorithmic}[1]
        \Function{F}{$n$}
            \If {$n = 0$}
                \State \Return $0$
            \Else
                \State \Return $\mathrm{e} -(n+1)F(n-1)$
            \EndIf
        \EndFunction
        \State $y_n \gets F(n)$
        \State $\varepsilon_n \gets \frac{|y_n - y_n^*|}{|y_n^*|}$
    \end{algorithmic}
\end{algorithm}
