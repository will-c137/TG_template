\section{测试算法伪代码}
\begin{algorithm}
\caption{测试算法伪代码}
\begin{algorithmic}
\REQUIRE $a, b, \epsilon$
\ENSURE $x^*$, 使得$|f(x^*)| < \epsilon$
\STATE $x^* = \frac{a + b}{2}$
\WHILE{$|f(x^*)| > \epsilon$}
\IF{$f(a) f(x^*) < 0$}
\STATE $b = x^*$
\ELSE
\STATE $a = x^*$
\ENDIF
\STATE $x^* = \frac{a + b}{2}$
\ENDWHILE
\end{algorithmic}
\end{algorithm}
