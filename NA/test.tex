\section{测试算法伪代码}
\begin{algorithm}
\caption{测试算法伪代码}
\begin{algorithmic}
\Require{$f(x, y)$}
\Ensure{$x_0, y_0, h, n$}
\State $x \leftarrow x_0$
\State $y \leftarrow y_0$
\For{$i=1$ to $n$}
\State $y \leftarrow y + h f(x, y)$
\State $x \leftarrow x + h$
\EndFor
\State \textbf{return} $x, y$
\end{algorithmic}
\end{algorithm}
